Click an AJAX call below and
result here

Get a new UID

$.ajax({
	method: 'PUT',
	url: 'rest.php',
	type: 'html',
	data: {'action': 'uid', 'data': 'employee'},
	success: function(data) {
		$("#result").html(data);
	},
	error: function(status){
		$("#result").html(status);
	}
});
